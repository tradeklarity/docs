---
title: 'Market Analytics'
openapi: openapi-api.yaml GET /markets/{slug}/analytics
description: Get aggregated holder counts, PnL, and trader types by outcome
---

## Path Parameters

<ParamField path="slug" type="string" required>
  Market slug identifier (e.g., `will-trump-win-2024`)
</ParamField>

## Response

<ResponseField name="holders" type="object">
  <ResponseField name="yes_holders" type="integer">
    Number of unique holders for Yes outcome
  </ResponseField>
  <ResponseField name="no_holders" type="integer">
    Number of unique holders for No outcome
  </ResponseField>
  <ResponseField name="total_holders" type="integer">
    Total unique holders (deduplicated)
  </ResponseField>
</ResponseField>

<ResponseField name="pnl" type="object">
  <ResponseField name="yes_holders_total_pnl" type="number">
    Sum of realized PnL for Yes holders (USD)
  </ResponseField>
  <ResponseField name="no_holders_total_pnl" type="number">
    Sum of realized PnL for No holders (USD)
  </ResponseField>
  <ResponseField name="total_pnl" type="number">
    Total realized PnL across all holders (USD)
  </ResponseField>
</ResponseField>

<ResponseField name="trader_types" type="object">
  <ResponseField name="yes" type="object">
    <ResponseField name="insider_holders" type="integer">
      Fresh wallet holders
    </ResponseField>
    <ResponseField name="smart_money_holders" type="integer">
      Smart money holders
    </ResponseField>
    <ResponseField name="whale_holders" type="integer">
      Portfolio whale holders
    </ResponseField>
  </ResponseField>
  <ResponseField name="no" type="object">
    <ResponseField name="insider_holders" type="integer">
      Fresh wallet holders
    </ResponseField>
    <ResponseField name="smart_money_holders" type="integer">
      Smart money holders
    </ResponseField>
    <ResponseField name="whale_holders" type="integer">
      Portfolio whale holders
    </ResponseField>
  </ResponseField>
</ResponseField>

<ResponseField name="meta" type="object">
  <ResponseField name="marketId" type="string">
    Condition ID
  </ResponseField>
  <ResponseField name="slug" type="string">
    Market slug
  </ResponseField>
</ResponseField>
