---
title: 'Whale Detection'
openapi: openapi-api.yaml GET /markets/{slug}/whales
description: Get whale holders for a market
---

## Overview

Returns holders classified as **whales** for a specific market. Whales are wallets with large portfolio exposure ($100k+ total portfolio or large position relative to market size).

<Card title="Holder Classification" icon="lightbulb" href="/concepts/holder-classification">
  Learn how whale, smart money, and insider classifications work.
</Card>

## Path Parameters

<ParamField path="slug" type="string" required>
  Market slug identifier (e.g., `will-trump-win-2024`)
</ParamField>

## Query Parameters

<ParamField query="lookbackHours" type="integer">
  Hours to look back from now. If not provided, uses market creation date.
</ParamField>

## Response

<ResponseField name="data" type="array">
  <Expandable title="Participant object">
    <ResponseField name="address" type="string">
      Wallet address
    </ResponseField>
    <ResponseField name="isPortfolioWhale" type="boolean">
      Whether holder is classified as a portfolio whale (always `true` in this endpoint)
    </ResponseField>
    <ResponseField name="isSmartMoney" type="boolean">
      Whether holder also meets smart money criteria
    </ResponseField>
    <ResponseField name="isInsider" type="boolean">
      Whether holder is also classified as an insider
    </ResponseField>
    <ResponseField name="winRatePct" type="number | null">
      Win rate percentage across closed positions (0-100), or null if unavailable
    </ResponseField>
    <ResponseField name="totalRealizedPnlUsd" type="number | null">
      Total realized PnL in USD, or null if unavailable
    </ResponseField>
    <ResponseField name="closedPositions" type="number | null">
      Number of closed/settled positions, or null if unavailable
    </ResponseField>
    <ResponseField name="portfolioValueUsd" type="number | null">
      Total portfolio value in USD, or null if unavailable
    </ResponseField>
    <ResponseField name="ageHours" type="number | null">
      Account age in hours on Polymarket, or null if unavailable
    </ResponseField>
    <ResponseField name="profile" type="object | null">
      Polymarket user profile (null if the user has no profile)
      <Expandable title="Profile fields">
        <ResponseField name="name" type="string | null">
          Display name
        </ResponseField>
        <ResponseField name="pseudonym" type="string | null">
          Pseudonym
        </ResponseField>
        <ResponseField name="displayName" type="string | null">
          Formatted display name
        </ResponseField>
        <ResponseField name="profileImage" type="string | null">
          Profile image URL
        </ResponseField>
        <ResponseField name="xUsername" type="string | null">
          X (Twitter) username
        </ResponseField>
        <ResponseField name="verifiedBadge" type="boolean">
          Whether profile has a verified badge
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="shares" type="number">
      Number of shares held in this market
    </ResponseField>
    <ResponseField name="positionValueUsd" type="number | null">
      Position value in USD (shares * price), or null if price unavailable
    </ResponseField>
    <ResponseField name="walletAgeHours" type="number | null">
      Wallet age in hours, or null if unavailable
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  <ResponseField name="marketId" type="string">
    Condition ID
  </ResponseField>
  <ResponseField name="slug" type="string">
    Market slug
  </ResponseField>
  <ResponseField name="window" type="object">
    <Expandable title="Window fields">
      <ResponseField name="since" type="string">
        Start of lookback window (ISO 8601)
      </ResponseField>
      <ResponseField name="until" type="string">
        End of lookback window (ISO 8601)
      </ResponseField>
    </Expandable>
  </ResponseField>
</ResponseField>
